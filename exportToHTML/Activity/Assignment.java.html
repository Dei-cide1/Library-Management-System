package Activity;

public class Car {
    String brand;
    String model;
    int year;
    double price;
    public Car(){
        System.out.println("Default Constructor Called");
        this.brand = "Unknown";
        this.model = "Unknown";
        this.year = 0;
        this.price = 0.00;
    }

    public Car(String brand, String model, int year, double price){
        this.brand = brand;
        this.model = model;
        this.year = year;
        this.price = price;
    }

    public Car(Car c){
        this.brand = c.brand;
        this.model = c.model;
        this.year = c.year;
        this.price = c.price;
    }

    public void displayInfo(){
        System.out.println("Brand: " + brand);
        System.out.println("Model: " + model);
        System.out.println("Year: " + year);
        System.out.println("Price: " + price);
        System.out.println();
    }
}
