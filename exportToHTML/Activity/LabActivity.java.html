package Activity;

import javax.swing.*;
import java.awt.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import java.io.IOException;

public class Activity {
    public static void main(String[] args) throws IOException {
        new Myframe().setVisible(true);
    }
}

package Activity;

import javax.swing.*;
        import java.awt.*;

public class Myframe extends JFrame {
    public Myframe() {
        this.setSize(500, 500);
        this.setTitle("User Entry Information Form");
        this.setLayout(new BorderLayout());
        this.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        this.setResizable(false);
        mainPanel panel = new mainPanel(this);
        this.add(panel.panel);
        this.pack();
        this.setVisible(true);
    }
}

package Activity;

import javax.swing.*;
        import java.awt.*;
        import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;

public class mainPanel implements ActionListener {
    JPanel panel;
    JButton submit;
    JButton clear;
    JButton exit;
    Myframe frame;
    JTextField fullname;
    JTextField age;
    JTextField address;
    JTextField email;
    JTextField number;

    public mainPanel(Myframe frame) {
        this.frame = frame;

        panel = new JPanel();
        panel.setPreferredSize(new Dimension(500, 500));
        panel.setBackground(Color.decode("#121318"));
        panel.setLayout(null);

        JLabel nameLabel = new JLabel("FulLname :");
        nameLabel.setForeground(Color.WHITE);
        nameLabel.setVerticalAlignment(JLabel.CENTER);
        nameLabel.setVerticalTextPosition(JLabel.CENTER);
        nameLabel.setBounds(20, 60, 100, 30);
        fullname = new JTextField();
        fullname.setBounds(160, 60, 300, 30);
        panel.add(nameLabel);
        panel.add(fullname);

        JLabel ageLabel = new JLabel("Age :");
        ageLabel.setForeground(Color.WHITE);
        ageLabel.setVerticalAlignment(JLabel.CENTER);
        ageLabel.setVerticalTextPosition(JLabel.CENTER);
        ageLabel.setBounds(20, 120, 100, 30);
        age = new JTextField();
        age.setBounds(160, 120, 300, 30);
        panel.add(ageLabel);
        panel.add(age);

        JLabel addressLabel = new JLabel("Address :");
        addressLabel.setForeground(Color.WHITE);
        addressLabel.setVerticalAlignment(JLabel.CENTER);
        addressLabel.setVerticalTextPosition(JLabel.CENTER);
        addressLabel.setBounds(20, 180, 100, 30);
        address = new JTextField();
        address.setBounds(160, 180, 300, 30);
        panel.add(addressLabel);
        panel.add(address);

        JLabel emailLabel = new JLabel("Email Address :");
        emailLabel.setForeground(Color.WHITE);
        emailLabel.setVerticalAlignment(JLabel.CENTER);
        emailLabel.setVerticalTextPosition(JLabel.CENTER);
        emailLabel.setBounds(20, 240, 150, 30);
        email = new JTextField();
        email.setBounds(160, 240, 300, 30);
        panel.add(emailLabel);
        panel.add(email);

        JLabel numberLabel = new JLabel("Contact Number :");
        numberLabel.setForeground(Color.WHITE);
        numberLabel.setVerticalAlignment(JLabel.CENTER);
        numberLabel.setVerticalTextPosition(JLabel.CENTER);
        numberLabel.setBounds(20, 300, 150, 30);
        number = new JTextField();
        number.setBounds(160, 300, 300, 30);
        panel.add(numberLabel);
        panel.add(number);

        submit = new JButton("Submit");
        submit.setFocusable(false);
        submit.setBounds(200, 350, 100, 30);
        submit.addActionListener(this);
        panel.add(submit);

        clear = new JButton("Clear");
        clear.setFocusable(false);
        clear.setBounds(200, 390, 100, 30);
        clear.addActionListener(this);
        panel.add(clear);

        exit = new JButton("exit");
        exit.setFocusable(false);
        exit.setBounds(200, 430, 100, 30);
        exit.addActionListener(this);
        panel.add(exit);
    }

    @Override
    public void actionPerformed(ActionEvent e) {
        if(e.getSource() == submit){
            JOptionPane.showMessageDialog(null, fullname.getText() + "\n" + age.getText() + "\n" + address.getText() + "\n" + email.getText() + "\n" + number.getText());
        }else if(e.getSource() == clear){
            fullname.setText("");
            age.setText("");
            address.setText("");
            email.setText("");
            number.setText("");
        }else if(e.getSource() == exit){
            frame.dispose();
        }
    }
}
